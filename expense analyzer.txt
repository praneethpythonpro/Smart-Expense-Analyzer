```python
# Smart Expense Analyzer
# Beginner to Intermediate Python Project

import pandas as pd
import matplotlib.pyplot as plt
import os

file_name = "expenses.csv"

# Create file if not exists
if not os.path.exists(file_name):
    df = pd.DataFrame(columns=["Date", "Type", "Category", "Amount"])
    df.to_csv(file_name, index=False)

def add_transaction():
    date = input("Enter Date (YYYY-MM-DD): ")
    t_type = input("Enter Type (Income/Expense): ").capitalize()
    category = input("Enter Category (Food, Travel, Salary, Shopping etc): ")
    amount = float(input("Enter Amount: "))

    new_data = pd.DataFrame([[date, t_type, category, amount]],
                            columns=["Date", "Type", "Category", "Amount"])

    new_data.to_csv(file_name, mode='a', header=False, index=False)
    print("Transaction Added Successfully!\n")

def show_summary():
    df = pd.read_csv(file_name)

    income = df[df["Type"] == "Income"]["Amount"].sum()
    expense = df[df["Type"] == "Expense"]["Amount"].sum()
    balance = income - expense

    print("\n------ Financial Summary ------")
    print("Total Income :", income)
    print("Total Expense:", expense)
    print("Balance      :", balance)

def category_analysis():
    df = pd.read_csv(file_name)

    expense_data = df[df["Type"] == "Expense"]
    category_sum = expense_data.groupby("Category")["Amount"].sum()

    print("\nExpense by Category:")
    print(category_sum)

    category_sum.plot(kind="bar", figsize=(8,5))
    plt.title("Expense Analysis")
    plt.xlabel("Category")
    plt.ylabel("Amount")
    plt.tight_layout()
    plt.show()

def saving_tip():
    df = pd.read_csv(file_name)

    expense_data = df[df["Type"] == "Expense"]
    if len(expense_data) == 0:
        print("No expense data available.")
        return

    top_category = expense_data.groupby("Category")["Amount"].sum().idxmax()
    print(f"\nYou spend most on {top_category}. Try reducing it to save more.")

while True:
    print("\n===== Smart Expense Analyzer =====")
    print("1. Add Transaction")
    print("2. Show Summary")
    print("3. Category Analysis")
    print("4. Saving Suggestion")
    print("5. Exit")

    choice = input("Enter Choice: ")

    if choice == "1":
        add_transaction()
    elif choice == "2":
        show_summary()
    elif choice == "3":
        category_analysis()
    elif choice == "4":
        saving_tip()
    elif choice == "5":
        print("Thank You!")
        break
    else:
        print("Invalid Choice. Try Again.")
```
